# Logistic Regression
```R
example = read.csv("examples.csv")
x = example$x
a = example$y.A

m = 100

# scatter plot
par(mfrow=c(1,2))
plot(x, a/m, xlim=c(0,100), ylim=c(0,1), ylab="p")
y = cbind(a, m-a)

# logistic regression
model = glm(y~x, family='binomial')
xnew = seq(0,100, length.out=100)
fitted = predict(model, newdata = data.frame(x=xnew), type="response")
lines(xnew,fitted, lwd=2, col="blue")

# residuals
residuals = resid(model, type="response")
plot(model$fitted.values, residuals, xlab="Fitted values", ylab="Residuals", ylim=c(-0.25,0.25))
abline(h=0, lty=2)

r.sd = sd(residuals)
abline(h= -2*r.sd, lty=2, col="red")
abline(h=2*r.sd, lty=2, col="red")
```
